Confirmation dialog customized with JS:

$("#complexConfirm").confirm({
    title:"Delete confirmation",
    text:"This is very dangerous, you shouldn't do it! Are you really really sure?",
    confirm: function(button) {
        alert("You just confirmed.");
    },
    cancel: function(button) {
        alert("You aborted the operation.");
    },
    confirmButton: "Yes I am",
    cancelButton: "No"
});

Manual triggering 
<button class="btn btn-primary" id="manualTrigger">Click me</button>
$("#manualTrigger").click(function() {
    $.confirm({
        text: "This is a confirmation dialog manually triggered! Please confirm:",
        confirm: function(button) {
            alert("You just confirmed.");
        },
        cancel: function(button) {
            alert("You cancelled.");
        }
    });
});
No cancellation button 
<button class="btn btn-primary" id="noCancelButton">Click me</button>
$("#noCancelButton").confirm({
    text: "This is a confirmation dialog manually triggered! Please confirm:",
    cancelButton: false,
    confirm: function() {
        alert("You just confirmed.");
    },
});

Nested 
<button class="btn btn-primary" id="nested">Click me</button>
$("#nested").confirm({
    title:"First confirmation",
    text: "Are you sure?",
    confirm: function(button) {
        $.confirm({
            title: "Second confirmation",
            text: "This is very dangerous, you shouldn't do it! Are you really really sure?‹br›‹i›“Only fools never change their mind.” Talleyrand‹/i›",
            confirm: function() {
                alert("You just confirmed.");
            },
            cancel: function() {
                alert("You cancelled. After first confirmation");
            },
            confirmButton: "Yes I am !",
            cancelButton: "No, it was a mistake",
            nested: true
        });
    },
    cancel: function(button) {
        alert("You cancelled.");
    },
    confirmButton: "Yes I am",
    cancelButton: "No"
});
